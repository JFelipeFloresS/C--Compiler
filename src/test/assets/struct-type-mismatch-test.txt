// Struct type mismatch tests

struct Point {
    int x;
    int y;
};

struct Vector {
    int x;
    int y;
};

void main() {
    struct Point p1;
    struct Vector v1;
    int i;

    // 1. Accessing a field on a non-struct variable.
    i.x = 10; // Error: 'i' is not a struct

    // 2. Accessing a non-existent field on a struct.
    p1.z = 20; // Error: 'z' is not a field of 'p1'

    // 3. Assigning a struct to a variable of a different type.
    i = p1; // Error: cannot assign a struct to an int

    // 4. Assigning a non-struct value to a struct variable.
    p1 = i; // Error: cannot assign an int to a struct

    // 5. Assigning a struct to another struct of a different definition.
    p1 = v1; // Error: cannot assign a Vector to a Point
}
